& \left(2^{a}+2^{-a}\right)^{2}=2^{2 a}+2^{-2 a}+2 \\
& 2^{a}+2^{-a}=3 \quad \frac{2^{3 a}+1}{2^{a}+2^{-a}}
\end{aligned}
$$
